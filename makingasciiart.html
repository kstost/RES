Converting an image to ASCII art involves a series of steps:

  1. Load the Image: Read the image file to understand its dimensions and color information.
  2. Grayscale Conversion: Convert the image to grayscale. This simplifies the information in the image to a single channel, making it easier to process.
  3. Resize the Image: Shrink the image to a smaller size to reduce the amount of data that needs to be processed. This is also necessary because ASCII characters are not square; they are taller than they are wide.
  4. Map Pixels to ASCII Characters: Loop through the pixels in the resized grayscale image and map each one to a corresponding ASCII character based on its brightness.
  5. Assemble ASCII Art: Create a string that represents the ASCII art and optionally save it to a text file.

import os
import zipfile

# Create a directory to store the ASCII art text files for each frame
ascii_art_frames_dir = '/mnt/data/ascii_art_frames'
os.makedirs(ascii_art_frames_dir, exist_ok=True)

# Step 1: Open Video File
cap = cv2.VideoCapture(video_path)
if not cap.isOpened():
    result = "Failed to open the video file."
else:
    # Step 2: Initialize Variables
    frame_number = 0
    frame_filenames = []

    # Step 3: Loop Through Frames
    while True:
        ret, frame = cap.read()
        if not ret:
            break  # Exit the loop if the video is over

        # Convert the frame to grayscale
        frame_gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

        # Resize the frame to make it 10 times smaller
        resized_frame = cv2.resize(frame_gray, (new_frame_width_10x, new_frame_height_10x))

        # Normalize and map pixels to ASCII characters
        resized_frame_normalized = resized_frame / 255.0
        ascii_map_frame = np.zeros_like(resized_frame_normalized, dtype=object)
        for i in range(len(ascii_characters)):
            ascii_map_frame[np.logical_and(resized_frame_normalized >= i / len(ascii_characters),
                                           resized_frame_normalized < (i + 1) / len(ascii_characters))] = ascii_characters[i]

        # Assemble the ASCII art
        ascii_art_frame = ""
        for row in ascii_map_frame:
            doubled_row = "".join([char * 2 for char in row])
            ascii_art_frame += doubled_row + "\n"

        # Save the ASCII art to a text file, with 3-digit frame number
        frame_filename = os.path.join(ascii_art_frames_dir, f"{frame_number:03d}.txt")
        with open(frame_filename, 'w') as f:
            f.write(ascii_art_frame)

        frame_filenames.append(frame_filename)
        frame_number += 1

    # Step 4: Close Video File
    cap.release()
