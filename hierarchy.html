import matplotlib.pyplot as plt
import networkx as nx

# Create a directed graph
G = nx.DiGraph()

# Add nodes with their descriptions
nodes = {
    'html': 'html',
    'head': 'head',
    'title': 'title',
    'body': 'body',
    'div': 'div',
    'human': 'human',
    'span': 'span'
}

G.add_nodes_from(nodes.keys())

# Add edges to represent the hierarchy
edges = [
    ('html', 'head'),
    ('html', 'body'),
    ('head', 'title'),
    ('body', 'div'),
    ('div', 'human'),
    ('human', 'span')
]

G.add_edges_from(edges)

# Draw the graph
pos = nx.spring_layout(G, seed=42)  # positions for all nodes
labels = {node: nodes[node] for node in G.nodes()}

nx.draw(G, pos, labels=labels, with_labels=True, node_color='lightblue', font_weight='bold', node_size=2000, font_size=20, arrows=True)
plt.title("HTML Document Hierarchy")
plt.show()
