from collections import Counter

# Split the 'Winning Numbers' into individual numbers and flatten the list
winning_numbers = lottery_data['Winning Numbers'].str.split(' ').explode()

# Count the occurrences of each number
winning_numbers_count = Counter(winning_numbers)

# Sort the numbers by frequency
sorted_winning_numbers = sorted(winning_numbers_count.items(), key=lambda x: x[1], reverse=True)

# Plot for most frequent winning numbers
axes[0].bar(winning_numbers, winning_counts, color='blue')
axes[0].set_title('Top 10 Most Frequent Winning Numbers')
axes[0].set_xlabel('Winning Numbers')
axes[0].set_ylabel('Frequency')
axes[0].set_xticks(np.arange(len(winning_numbers)))

# Plot for most frequent Mega Ball numbers

axes[1].set_ylabel('Frequency')
