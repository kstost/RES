    worksheet = workbook.add_worksheet(country)
    
    # Write headers
    worksheet.write('A1', 'Rank')
    worksheet.write('B1', 'Youtuber')
    worksheet.write('C1', 'Subscribers')
    
    # Write data
    for i, (index, row) in enumerate(country_df.iterrows()):
        worksheet.write(i + 1, 0, row['rank'])
        worksheet.write(i + 1, 1, row['Youtuber'])
        worksheet.write(i + 1, 2, row['subscribers'])

    chart = workbook.add_chart({'type': 'column'})
    
    # Configure the chart
    chart.add_series({
        'name':       f'{country} Subscribers',

        'values':     [country, 1, 2, len(country_df), 2],
