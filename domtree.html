# Importing the libraries to create a graphical representation of the DOM tree
import matplotlib.pyplot as plt
import networkx as nx

# Initialize the graph
G = nx.DiGraph()

# Define the nodes and edges based on the DOM tree structure
nodes = [
    'html', 'head', 'title', 'style', 'body', 'nav', 'ul', 'li', 'a',
    'div', 'h1', 'p', 'table', 'thead', 'tr', 'th', 'tbody', 'td', 'footer'
]
edges = [
    ('html', 'head'), ('head', 'title'), ('head', 'style'),
    ('html', 'body'), ('body', 'nav'), ('nav', 'ul'), ('ul', 'li'), ('li', 'a'),
    ('body', 'div'), ('div', 'h1'), ('div', 'p'), ('div', 'table'), ('table', 'thead'), ('thead', 'tr'),
    ('tr', 'th'), ('table', 'tbody'), ('tbody', 'tr'), ('tr', 'td'), ('body', 'footer'), ('footer', 'a')
